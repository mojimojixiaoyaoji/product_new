  "c-kw">local name="$1" url="$2" expect="${3:-200}"

  "c-kw">else
    "c-kw">echo -e "  ${RED}✗${NC} ${name} (${url}) → HTTP ${code}"
    ((fail++)) || "c-kw">true
  "c-kw">fi
